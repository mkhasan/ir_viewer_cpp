> {
> 	try
> 	{
> 		// check format
> 		if(bpsample != 8 && bpsample != 16)

> 			throw -1;
> 		if(channels < 1 || channels > 2)
> 			throw -1;
> 
> 		int buffersize = bpsample * sampling * channels / 8 * 1;
> 
> 		TRACE("BufferSize %d\n", buffersize);
> 		m_pSpkBuffer = new CCircularBuffer(buffersize);

> 			throw -1;

> 		{
